with open(os.path.join(csv_path, csv_file)) as f:
    csv_reader = csv.reader(f, delimiter=',')
    line_count = 0
    for row in csv_reader:
        print(row[0])
        resp = client.create_segmentannotation_task(
            #project='lab_mussel_segmentation',
            #batch='Lab-0',
            urgency='standard', #'immediate',
            callback_url='gallowaa@uoguelph.ca',
            instruction='Draw a tight polygon around the live **mussels** in the image. Ignore empty white shells.',
            attachment_type='image',
            attachment=row[0],
            labels=[
              'background',
              'mussel',
            ],
            allow_unlabeled=False
        )
        print(resp)
        line_count += 1
    print(f'Processed {line_count} lines.')
